利用Dockerfile

FROM centos:latest
MAINTAINER walljay <760813193@qq.com>
RUN yum -y update
RUN yum -y install sudo
RUN yum -y install net-tools
RUN yum -y install openssh-server
RUN yum -y install vim
RUN yum -y install java-1.8.0-openjdk-devel
CMD ["/bin/bash"]
